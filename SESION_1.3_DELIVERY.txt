         "tasa_error_pct": 2.1,
         "picks_7d": 450,
         "bultos_7d": 1050,
         "vel_7d": 24.5
       },
       ...
     ]
   }

2. GET /api/comparativas/clusters

   Retorna: Clustering automático en 3 grupos

   Response:
   {
     "clusters": {
       "high_performers": {
         "count": 3,
         "operarios": [...],
         "stats": {
           "avg_score": 85.5,
           "avg_picks": 5200,
           "avg_vel": 24.3,
           "avg_error": 2.1
         }
       },
       "mid_range": {...},
       "learning": {...}
     },
     "all_scored": [...]
   }

3. GET /api/comparativas/benchmark/{operario_id}

   Retorna: Comparativa individual vs grupo promedio

   Response:
   {
     "operario": {
       "total_picks": 5200,
       "velocidad_promedio": 24.3,
       "tasa_error_pct": 2.1
     },
     "grupo": {
       "total_picks_promedio": 3850,
       "velocidad_promedio": 28.5,
       "tasa_error_pct": 3.2
     },
     "comparativa": {
       "picks_diferencia_pct": 35.1,
       "velocidad_diferencia_pct": 14.7,
       "error_diferencia_pct": -34.4
     }
   }

   - Datos siempre actualizados

================================================================================
                         CLUSTERING ALGORITHM
================================================================================

Score = (total_picks / velocidad_promedio) × (1 - error_rate)

Percentile-Based Division:
├─ High Performers:  Top 33% by score (verde)
├─ Mid Range:        Middle 34% (amarillo)
└─ Learning:         Bottom 33% (rojo)

Benchmark Calculation:
├─ picks_diff = ((op - group) / group) × 100
├─ vel_diff = ((group - op) / group) × 100  [negativo = mejor]
└─ error_diff = ((op - group) / group) × 100  [negativo = mejor]
